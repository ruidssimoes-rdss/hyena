  <SheetTrigger>
    <Button>Open Sidebar</Button>
  </SheetTrigger>

    <SheetHeader>
      <SheetTitle>Sidebar</SheetTitle>

  </SheetContent>

## Gesture Dismissal

The sheet can be dismissed by:
- Swiping in the dismiss direction (down for bottom, up for top, etc.)
- Tapping the backdrop
- Using the `SheetClose` component

## Controlled

```tsx
const [open, setOpen] = useState(false)

<Sheet open={open} onOpenChange={setOpen}>
  <SheetTrigger>
    <Button>Open</Button>
  </SheetTrigger>
  <SheetContent>
    <SheetHeader>
      <SheetTitle>Controlled Sheet</SheetTitle>
    </SheetHeader>
    <Button onPress={() => setOpen(false)}>Close</Button>
  </SheetContent>
</Sheet>
```

## Installation

```bash
npx r-ui add sheet
```

## Usage

```tsx
import {
  Sheet,
  SheetTrigger,
  SheetContent,
  SheetHeader,
  SheetTitle,
  SheetDescription,
  SheetFooter,
  SheetClose,
} from '@r-ui/react-native'
import { Button } from '@r-ui/react-native'

export default function MyComponent() {
  return (
    <Sheet>
      <SheetTrigger>
        <Button>Open Sheet</Button>
      </SheetTrigger>
      <SheetContent>
        <SheetHeader>
          <SheetTitle>Sheet Title</SheetTitle>
          <SheetDescription>
            This is a description of the sheet content.
          </SheetDescription>
        </SheetHeader>
        <SheetFooter>
          <SheetClose>
            <Button>Done</Button>
          </SheetClose>
        </SheetFooter>
      </SheetContent>
    </Sheet>
  )
}
```

## Props

### Sheet

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | - | Controlled open state |
| `onOpenChange` | `(open: boolean) => void` | - | Callback when open state changes |
| `side` | `'bottom' \| 'top' \| 'left' \| 'right'` | `'bottom'` | Side to show sheet |

### SheetTrigger

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactElement` | - | Trigger element |
| `style` | `ViewStyle` | - | Additional styles |

### SheetContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Sheet content |
| `style` | `ViewStyle` | - | Additional styles |

### SheetHeader

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Header content |
| `style` | `ViewStyle` | - | Additional styles |

### SheetFooter

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Footer content |
| `style` | `ViewStyle` | - | Additional styles |

### SheetTitle

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Title content |

### SheetDescription

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Description content |

### SheetClose

| Prop | Type | Default | Description |
